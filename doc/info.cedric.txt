	DAL	= Data Acces Layer
		PDO
	
	DAO = Data Access Object
		Table <=> liste d'objet
		Un enregistrement de table <=> à un objet
		Un champ d'enregistrement <=> à un attribut
		Valeur d'un champ <=> Valeur d'un attribut d'objet
	
	ORM = Object Relation Mapping
		Gere les dépendances entre les objets (clé étrangère)

			// Création d'un forum
			$forum = new Forum('Toto', 'Titre Forum');
			$forumId = $forum->getId();
			
			// Création d'un sujet dans le forum
			$sujet = new Sujet('Toto', 'Titre du Sujet', $forumId);
			$sujetId = $sujet->getId();
			
			// Création d'un message dans le sujet
			$message = new Message('Toto', 'Titre Message', 'Corps du message', $sujetId); 
			$messageId = $message->getId();
			
			// Affichage d'un seul message dans une page
			$message->getMessage($messageId);
		

		
		
		/*
		*	Réalisation
		*
		*/
		
		    Une URL.
			Un module.
			Une action.
			Un tableau comportant les noms des variables.
			Un tableau clé/valeur comportant les noms/valeurs des variables.
	
	
		Route possible
			?module=X&action=Y			&var=Z
			?page=MODULE&action=ACTION	&var=Z
			
			?page=forum&action=forumId	&var=Z
			?page=forum&action=forumId	&var=Z
			
		
			?page=forum&forum=id	&page=2
			?page=forum&sujet=id	&page=2
			?page=forum&message=id	&page=2
			
			?page=forum&action=NouveauForum
			?page=forum&action=NouveauSujet				&var=z
			?page=forum&action=NouveauMessage			&var=z
			
			?page=forum&action=ModificationForum		&var=z
			?page=forum&action=ModificationSujet		&var=z		
			?page=forum&action=ModificationMessage		&var=z


		Route sélectionnées
			?page=forum&action=ajoutForum		&var=z
				Si var != '', ajoute un nouveau forum
				Si var == X, modifie un forum
			
			?page=forum&action=sujet			&var=z
				Affiche l'ensemble des sujets d'un forum, var correspond a un forum
				
			?page=forum&action=ajoutSujet		&var=z
				
			?page=forum&action=ajoutMessage		&var=z
				Modifie un message, var == l'Id du sujet
			
			?page=forum&action=modifieMessage		&var=z
				Si var != '', ajoute un nouveau message
				Si var == X, modifie un message


		Route sélectionnées - Résumé
			action = ajoutForum 	& var = z
			action = sujet 			& var = z
			action = ajoutSujet		& var = z
			action = ajoutMessage 	& var = z	
			action = modifieMessage	& var = z
